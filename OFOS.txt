CREATE TABLE CUSTOMER
(
CUSTOMER_ID NUMBER(10),
FIRST_NAME  VARCHAR2(20),
LAST_NAME   VARCHAR2(20),
REGISTRATION_DATE DATE,
EMAIL_ID VARCHAR2(50),
CONTACT_NO VARCHAR2(15),
PASSWORD VARCHAR2(40),
PRIMARY KEY ( CUSTOMER_ID )
);


CREATE TABLE EMPLOYEE
(
EMPLOYEE_ID NUMBER(10),
EMPLOYEE_NAME VARCHAR2(40),
HIRE_DATE DATE,
AVAILABILITY INT,
PRIMARY KEY ( EMPLOYEE_ID )
);


CREATE TABLE AREA
(
AREA_NAME VARCHAR2(15),
AREA_ID NUMBER(10),
DISTRICT VARCHAR2(15),
POSTAL_CODE INT,
PRIMARY KEY ( AREA_ID )
);


CREATE TABLE RESTAURENT
(
RESTAURENT_ID NUMBER(10),
RESTAURENT_NAME VARCHAR2(20),
OPEN_TIME VARCHAR2(4),
CLOSE_TIME VARCHAR2(4),
WEEKLY_LEAVE VARCHAR2(10),
AREA_ID NUMBER(10),
PRIMARY KEY ( RESTAURENT_ID ),
FOREIGN KEY ( AREA_ID )
	REFERENCES AREA ( AREA_ID )
);


CREATE TABLE ORDER
(
ORDER_ID NUMBER(10),
EMPLOYEE_ID NUMBER(10),
FOOD_ID NUMBER(10),
PRICE INT,
ORDER_TIME VARCHAR2(15),
DELIVERY_TIME VARCHAR2(15),
ADDRESS VARCHAR2(20),
PAYMENT_MATHOD VARCHAR2(10),
PRIMARY KEY ( ORDER_ID ),
FOREIGN KEY ( EMPLOYEE_ID )
	REFERENCES EMPLOYEE ( EMPLOYEE_ID )
);


CREATE TABLE FOOD
(
FOOD_NAME VARCHAR2(20),
CATEGORY VARCHAR2(20),
FOOD_ID NUMBER(10),
PRICE INT,
ORDER_ID NUMBER(10),
DURATION INT,
RESTAURENT_ID NUMBER(10),
PRIMARY KEY ( FOOD_ID ),
FOREIGN KEY ( ORDER_ID )
	REFERENCES ORDER ( ORDER_ID ),
FOREIGN KEY ( RESTAURENT_ID )
	REFERENCES RESTAURENT ( RESTAURENT_ID )
);



CREATE TABLE REVIEW
(
COMMENT VARCHAR2(1000),
RATING INT,
REVIEW_ID NUMBER(10),
RESTAURENT_ID NUMBER(10),
CUSTOMER_ID NUMBER(10),
PRIMARY KEY ( REVIEW_ID ),
FOREIGN KEY ( RESTAURENT_ID )
	REFERENCES RESTAURENT ( RESTAURENT_ID ),
FOREIGN KEY ( CUSTOMER_ID )
	REFERENCES CUSTOMER ( CUSTOMER_ID )
);



CREATE TABLE DELIVERED_ORDER
(
CUSTOMER_ID NUMBER(10),
ORDER_ID NUMBER(10),
PRIMARY KEY ( CUSTOMER_ID, ORDER_ID ),
FOREIGN KEY ( CUSTOMER_ID )
	REFERENCES CUSTOMER ( CUSTOMER_ID ),
FOREIGN KEY ( ORDER_ID )
	REFERENCES ORDER ( ORDER_ID )
);
